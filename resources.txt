Heroku commands:
Update heroku repo = 'git add --all' THEN 'git commit -am "commit message goes here"' THEN 'git push heroku master'
Drop database = 'heroku pg:reset DATABASE' OR 'heroku run rake db:reset' THEN 'heroku restart'
Recreate empty database = 'heroku run rake db:migrate'
